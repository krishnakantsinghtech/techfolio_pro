Performance Optimization
// Optimized data aggregation query
SELECT date_trunc('hour', timestamp) as hour,
  AVG(co2_level) as avg_co2,
  COUNT(*) as readings
FROM sensor_data
WHERE timestamp >= NOW() - INTERVAL '24 hours'
GROUP BY hour ORDER BY hour;
